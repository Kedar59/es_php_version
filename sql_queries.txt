CREATE TABLE user(
    ID int NOT NULL AUTO_INCREMENT PRIMARY KEY,
    username varchar(191),
    email varchar(191),
    password varchar(255),
    UNIQUE (email)
)
CREATE TABLE questions(
    ID int NOT NULL AUTO_INCREMENT PRIMARY KEY,
    title varchar(191),
    url varchar(191),
    category varchar(191),
    difficulty varchar(191),
    languages varchar(191)
)
